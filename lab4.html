Завдання №1

Умова

Зберегти введене значення у змінну value. Вивести це значення у Console. Перевірити: введене значення від'ємне, додатнє чи дорівнює нулю. Відповідь вивести за допомогою alert().

Код

let value = Number(prompt("Введіть число"));

console.log(`Введене число: ${value}`);

if (value > 0) {
    alert("Число додатнє");
} else if (value < 0) {
    alert("Число від'ємне");
} else {
    alert("Число дорівнює нулю");
}

Завдання №2

Умова

Змінна number може набувати значень 1, 2, 3 або 4. Якщо значення дорівнює 1 — зима, 2 — весна, 3 — літо, 4 — осінь. Розв'язати через switch-case.

Код

let number = prompt("Введіть число від 1 до 4");

let result;

switch (number) {
    case "1":
        result = "Зима";
        break;
    case "2":
        result = "Весна";
        break;
    case "3":
        result = "Літо";
        break;
    case "4":
        result = "Осінь";
        break;
    default:
        result = "Некоректне значення";
}

console.log(result);

Завдання №3

Умова

Є користувачі Admin та User з відповідними паролями. Запитати логін користувача через prompt(). Якщо логін правильний — запросити пароль. При успішній авторизації вивести привітання.

Код

const adminPassword = "admin123";
const userPassword = "user123";

let login = prompt("Введіть логін");

console.log(`Логін: ${login}`);

if (login === "Admin") {

    let password = prompt("Введіть пароль");

    if (password === adminPassword) {
        alert("Hello, Admin");
    } else {
        alert("Невірний пароль");
    }

} else if (login === "User") {

    let password = prompt("Введіть пароль");

    if (password === userPassword) {
        alert("Hello, User");
    } else {
        alert("Невірний пароль");
    }

} else if (login === "" || login === null) {

    alert("Cancelled");

} else {

    alert("I don't know you");

}

Завдання №5

Умова

Функція checkForSpam(message) перевіряє рядок на наявність слів spam або sale незалежно від регістру. Якщо слово знайдено — повертає true, інакше — false.

Код

function checkForSpam(message) {

    const normalizedMessage = message.toLowerCase();

    return normalizedMessage.includes("spam") ||
           normalizedMessage.includes("sale");
}

console.log(checkForSpam("Latest technology news"));
console.log(checkForSpam("Big SALE today"));
console.log(checkForSpam("Get best SPAM offers"));

Завдання №6

Умова

Написати функцію filterArray(numbers, value), яка повертає новий масив лише з тих елементів, що більші за value.

Код

function filterArray(numbers, value) {

    const result = [];

    for (let number of numbers) {

        if (number > value) {
            result.push(number);
        }

    }

    return result;
}

console.log(filterArray([1, 2, 3, 4, 5], 3));
console.log(filterArray([12, 24, 8, 41, 76], 20));

Завдання №7 (Варіант 7)

Умова 1

Створити одновимірний масив. Знайти найбільший серед від'ємних та найменший серед додатних елементів масиву. Застосувати функції.

Код

function findValues(arr) {

    let maxNegative = -Infinity;
    let minPositive = Infinity;

    for (let num of arr) {

        if (num < 0 && num > maxNegative) {
            maxNegative = num;
        }

        if (num > 0 && num < minPositive) {
            minPositive = num;
        }
    }

    return {
        maxNegative,
        minPositive
    };
}

let arr = [-8, 12, -3, 7, 2, -15, 9];

console.log("Масив:", arr);

const result = findValues(arr);

console.log("Найбільший серед від'ємних:", result.maxNegative);
console.log("Найменший серед додатних:", result.minPositive);

Умова 2

Упорядкувати масив у порядку зменшення методом вибору. Надрукувати вхідний та вихідний масив.

Код

function selectionSortDesc(arr) {

    for (let i = 0; i < arr.length - 1; i++) {

        let maxIndex = i;

        for (let j = i + 1; j < arr.length; j++) {

            if (arr[j] > arr[maxIndex]) {
                maxIndex = j;
            }
        }

        [arr[i], arr[maxIndex]] =
            [arr[maxIndex], arr[i]];
    }

    return arr;
}

let numbers = [-8, 12, -3, 7, 2, -15, 9];

console.log("Початковий масив:", numbers);

console.log(
    "Відсортований масив:",
    selectionSortDesc([...numbers])
);

Завдання №8

Умова

Створити двовимірний масив додатніх та від'ємних чисел. Сформувати окремо масив додатніх та від'ємних чисел. Замінити третій елемент додатного масиву введеним користувачем від'ємним числом.

Код

let matrix = [];

for (let i = 0; i < 4; i++) {

    let row = [];

    for (let j = 0; j < 4; j++) {

        row.push(
            Math.floor(Math.random() * 41) - 20
        );
    }

    matrix.push(row);
}

console.log("Двовимірний масив:");
console.table(matrix);

let positive = [];
let negative = [];

for (let row of matrix) {

    for (let item of row) {

        if (item >= 0) {
            positive.push(item);
        } else {
            negative.push(item);
        }
    }
}

console.log("Додатні числа:", positive);
console.log("Від'ємні числа:", negative);

let newValue = Number(
    prompt("Введіть від'ємне число")
);

positive[2] = newValue;

console.log("Оновлений масив додатних:");
console.log(positive);
