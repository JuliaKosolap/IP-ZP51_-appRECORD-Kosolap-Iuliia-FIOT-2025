<button onclick="showMessage()">
    SHOW ME
</button>

function showMessage() {
    const value =
        document.getElementById(
            "messageInput"
        ).value;

    console.log(
        "Введене значення:",
        value
    );
}

Завдання 3

Умова

Кнопка «Приховати» ховає текст, замінюючи його на зірочки. При повторному натисканні текст знову стає видимим.

Код

function togglePassword() {

    const input =
        document.getElementById(
            "passwordInput"
        );

    const button =
        document.getElementById(
            "toggleBtn"
        );

    if (input.type === "password") {

        input.type = "text";
        button.textContent =
            "Приховати";

    } else {

        input.type = "password";
        button.textContent =
            "Розкрити";
    }
}

    category => {

Умова

Змінювати колір фону сторінки на випадковий при натисканні кнопки.

Код

function getRandomHexColor() {
    return "#" +
        Math.floor(
            Math.random() * 16777215
        )
        .toString(16)
        .padStart(6, "0");
}

function changeColor() {

    const color =
        getRandomHexColor();

    document.body.style.backgroundColor =
        color;

    document.getElementById(
        "colorValue"
    ).textContent = color;
}

Висновки

Під час виконання лабораторної роботи було опрацьовано роботу з DOM-деревом, подіями браузера та механізмом делегування подій. Було створено інтерактивні елементи інтерфейсу, реалізовано обробку подій користувача, роботу з формами, зміною стилів та динамічним створенням елементів сторінки.

Отримано практичні навички використання об'єкта події, роботи з DOM-елементами, делегування подій та створення динамічних веб-інтерфейсів засобами JavaScript.
